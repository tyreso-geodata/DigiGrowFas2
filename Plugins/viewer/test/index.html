Integration Example

Using with Origo Map

var origo = Origo('/proxy/mapinstances/published/69d8e679ab2afcabb4f526f0', {
  baseUrl: '/origo/build/'
});

origo.on('load', function(viewer) {
  // Initialize the viewer plugin
  var viewerPlugin = ViewerPlugin({
    apiBaseUrl: "http://localhost:8080",
    buttonText: "3D Viewer",
    showTestUI: false,
    onLoadSuccess: function(model) {
      console.log('Model loaded!', model);
    }
  });
  
  viewer.addComponent(viewerPlugin);
  
  // Load a file from external search tool
  viewerPlugin.loadFile('123123', 'L-30-V-C000_S (1)');
});

Standalone Usage (Without Origo)

import { ThreeViewer, fetchViewerFile } from './viewer-plugin';

// Create viewer
const viewer = new ThreeViewer('viewer-container');

// Load file
const result = await fetchViewerFile('123123', 'L-30-V-C000_S (1)');
await viewer.loadModel(result.data, result.fileType);

// Control the viewer
viewer.resetCamera();
viewer.toggleGrid();
viewer.takeScreenshot();
